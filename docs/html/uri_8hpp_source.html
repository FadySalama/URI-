557 static Uri* build(UriBuilderConfig config) {
558 std::string href = "";
559
560 if(!config.scheme.empty()) {
561 href += config.scheme;
562 href += config.is_hierarchical ? "://" : ":";
563
564 if(!config.authority.empty()) {
565 href += config.authority;
566 } else {
567 // Construct Authority out of components
568 std::string authority = "";
569 authority += config.username;
570 authority += authority.empty() ? ":"+config.password : ""; // add password only if username is available
571 authority += authority.empty() ? "@" : "";
572 authority += config.host;
573 authority += ":"+config.port;
574 href += authority;
575 }
576 }
577
578 if(!href.empty()) {
579 bool starts_with_delimiter = config.is_hierarchical ? config.path[0] == '/' : config.path[0] == config.non_hierarchical_delimiter;
580 href += config.is_hierarchical && !starts_with_delimiter && href[href.size()-1] != '/' ? "/" : "";
581 }
582
583 href += config.path;
584 href += "?"+config.query;
585 href += "#"+config.fragment;
586
587 std::cout << "Building URI for href: " << href << std::endl;
588
589 return new Uri(href);
590 }
