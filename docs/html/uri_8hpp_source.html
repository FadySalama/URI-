492 class UriBuilder {
493 public:
494 static Uri* build(UriBuilderConfig config) {
495 std::string href = "";
496
497 if(!config.scheme.empty()) {
498 href += config.scheme;
499 href += config.is_hierarchical ? "://" : ":";
500
501 if(!config.authority.empty()) {
502 href += config.authority;
503 } else {
504 // Construct Authority out of components
505 std::string authority = "";
506 authority += config.username;
507 authority += authority.empty() ? ":"+config.password : ""; // add password only if username is available
508 authority += authority.empty() ? "@" : "";
509 authority += config.host;
510 authority += ":"+config.port;
511 href += authority;
512 }
513 }
514
515 if(!href.empty()) {
516 bool starts_with_delimiter = config.is_hierarchical ? config.path[0] == '/' : config.path[0] == config.non_hierarchical_delimiter;
517 href += config.is_hierarchical && !starts_with_delimiter && href[href.size()-1] != '/' ? "/" : "";
518 }
519
520 href += config.path;
521 href += "?"+config.query;
522 href += "#"+config.fragment;
523
524 std::cout << "Building URI for href: " << href << std::endl;
525
526 return new Uri(href);
527 }
